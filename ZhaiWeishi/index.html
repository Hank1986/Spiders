Export Cases to Excel

Click the button to login, fetch cases and details, and export to an Excel file.

import { runExport } from './exportCases.js'; const btn = document.getElementById('exportBtn'); const logEl = document.getElementById('log'); function log(msg) { const time = new Date().toLocaleTimeString(); logEl.textContent += `[${time}] ${msg}\n`; } btn.addEventListener('click', async () => { btn.disabled = true; log('Starting export...'); try { const result = await runExport({ perPage: 50, filename: 'cases.xlsx' }); log(`Export finished. Rows: ${result.count}`); alert('Export completed.'); } catch (e) { console.error(e); log(`Export failed: ${e?.message || e}`); alert(`Export failed: ${e?.message || e}`); } finally { btn.disabled = false; } });
